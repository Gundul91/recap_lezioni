      Parametri di default:
      function greet(name = 'Student', greeting = 'Welcome') {
        return `${greeting} ${name}!`;
      }
      I valori inseriti nelle tonde sono i valori di default nel caso non vengano passati quando viene chiamata la funzione

      greet(); // Welcome Student!
      greet('James'); // Welcome James!
      greet('Richard', 'Howdy'); // Howdy Richard!

      Parametri di default con array:
      function createGrid([width = 5, height = 5] = []) {
        return `Generates a ${width} x ${height} grid`;
      }
      il "= []" serve come array vuota di default in caso venga chiamata la funzione senza parametri

      Parametri di default con oggetti:
      function houseDescriptor({houseColor = 'green', shutterColors = ['red']} = {}) {
        return `I have a ${houseColor} house with ${shutterColors.join(' and ')} shutters`;
      }
      il "= {}" serve come oggetto vuota di default in caso venga chiamata la funzione senza parametri

      Sottoclasse:
      class Tree {
        constructor(size = '10', leaves = {spring: 'green', summer: 'green', fall: 'orange', winter: null}) {
          this.size = size;
          this.leaves = leaves;
          this.leafColor = null;
        }

        changeSeason(season) {
          this.leafColor = this.leaves[season];
          if (season === 'spring') {
            this.size += 1;
          }
        }
      }

      class Maple extends Tree {
        constructor(syrupQty = 15, size, leaves) {
          super(size, leaves);
          super(...) richiama il costruttore della classe parente
          this.syrupQty = syrupQty;
        }
        extend indica che è una sottoclasse di Tree

        changeSeason(season) {
          super.changeSeason(season);
          if (season === 'spring') {
            this.syrupQty += 1;
          }
        }
        super.changeSeason(...) richiama il metodo della classe parente

        gatherSyrup() {
          this.syrupQty -= 3;
        }
      }

      const myMaple = new Maple(15, 5);
      myMaple.changeSeason('fall');
      myMaple.gatherSyrup();
      myMaple.changeSeason('spring');
      
      IMPORTANTE: Prima di usare il this nella sottocalsse deve essere richiamato almeno una volta il "super"
