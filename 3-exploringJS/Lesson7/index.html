      Object.keys e Object.values:
      const dictionary = {
        car: 'automobile',
        apple: 'healthy snack',
        cat: 'cute furry animal',
        dog: 'best friend'
      };

      Object.keys(dictionary);
      // [car, apple, 'cat', 'dog']
      Object.keys restituisce un array di stringhe con le chiavi

      Object.values(dictionary);
      // ['automobile', 'healthy snack', 'cute furry animal', 'best friend']
      Object.values restituisce un array con i valori
